void obstacle :: obstacle_animation()
{
    
    float dT = GetFrameTime();
    if(this -> running_time < this -> frame_duration)
    {
        this -> running_time +=dT;
    }
    else
    {
        part_of_texture += 1;
        if(this -> part_of_texture > 5)
        {
                part_of_texture = 0;
        }
        this -> obstacle_rec.x = this -> obstacle_rec.width* this -> part_of_texture; 
        this -> running_time = 0;
    }
}
